163 class SPIFailureChecker
164  : public NewTestPartResultCheckHelper::Collector<NoTestPartResultReporter>
165 {
166 public:
167  explicit SPIFailureChecker(TestPartResult::Type type) : m_Type(type) {}
168 public:
169  AssertionResult GetResult(const ::std::string& substr)
170  {
171  const char* expected =
172  (m_Type == TestPartResult::kFatalFailure) ? "1 fatal failure" : "1 non-fatal failure";
173  const size_t num = count();
174  if( num != 1 )
175  {
176  AssertionResult ar = AssertionFailure() << "error: Expected: " << expected
177  << "\n Actual: " << num << " failures\n";
178  for( size_t i=0; i < num; ++i )
179  {
180  ar << GetTestPartResult(i);
181  }
182  return ar;
183  }
184 
185  const TestPartResult& tr = GetTestPartResult(0);
186  if( tr.type() != m_Type )
187  {
188  return AssertionFailure() << "error: Expected: " << expected
189  << "\"\n Actual:\n" << tr;
190  }
191 
192  if( strstr(tr.message(), substr.c_str()) == NULL )
193  {
194  return AssertionFailure() << "error: Expected: " << expected
195  << " containing \"" << substr
196  << "\"\n Actual:\n" << tr;
197  }
198  return AssertionSuccess();
199  }
200 private:
201  TestPartResult::Type m_Type;
202 };
