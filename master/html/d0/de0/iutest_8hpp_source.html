1892  }

1896  {
1897  if( TestEnv::is_output_option_dirty() )
1898  {
1899  if( TestEnv::has_output_option() )
1900  {
1901  TestEnv::flush_output_option();
1902 
1903  do
1904  {
1905 #if defined(__WANDBOX__)
1906  if( StderrXmlGeneratorListener::SetUp() )
1907  {
1908  break;
1909  }
1910  if( StderrJunitXmlGeneratorListener::SetUp() )
1911  {
1912  break;
1913  }
1914 #else
1915  if( DefaultXmlGeneratorListener::SetUp() )
1916  {
1917  break;
1918  }
1919  if( JunitXmlGeneratorListener::SetUp() )
1920  {
1921  break;
1922  }
1923 #endif
1924  IUTEST_LOG_(WARNING) << "unrecognized output format \"" << TestEnv::get_output_option() << "\" ignored.";
1925  } while( detail::AlwaysFalse() );
1926  }
1927  else

1929  TestEnv::event_listeners().Release(TestEnv::event_listeners().default_xml_generator());
