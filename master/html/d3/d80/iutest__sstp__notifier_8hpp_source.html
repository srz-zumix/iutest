76  {
77  m_code = -1;
78  if( !detail::IsStringForwardMatching(str, "SSTP/") ) {
79  return false;
80  }
81  char* p=NULL;
82  m_major = strtol(str+5, &p, 10);
83  m_minor = strtol(p+1, &p, 10);
84  m_code = strtol(p+1, &p, 10);
85  m_message = p;
86  return true;
87  }

212  str += "notranslate";
213  }
214  return "";
215  }
