34 class TestMenu
35 {
36  typedef ::std::map<WORD, const ::iutest::TestInfo*> TestInfoMap;
37  typedef ::std::map<WORD, const ::iutest::TestCase*> TestCaseMap;
38  WORD m_nID;
39  const WORD m_nIDTop;
40  TestInfoMap m_TestInfoList;
41  TestCaseMap m_TestCaseList;
42 public:
43  explicit TestMenu(WORD nIDTop) : m_nIDTop(nIDTop), m_nID(nIDTop) {}
44 public:
45  bool Create(HMENU hMenu)
46  {
47  if( hMenu == NULL )
48  {
49  return false;
50  }
51 
52  // テストを列挙
53  HMENU hRoot = AppendPopup(hMenu, "TestList");
54  if( hRoot == NULL )
55  {
56  return false;
57  }
58 
59  Append(hRoot, "以下をすべて実行", m_nID);
60  ++m_nID;
61 
62  ::iutest::UnitTest* pUnitTest = ::iutest::UnitTest::GetInstance();
63  const int testcase_count = pUnitTest->total_test_case_count();
64  for( int i=0; i < testcase_count; ++i )
65  {
66  const ::iutest::TestCase* pTestCase = pUnitTest->GetTestCase(i);
67  const int test_count = pTestCase->total_test_count();
68  HMENU hTestCase = AppendPopup(hRoot, pTestCase->name());
69  Append(hTestCase, "以下をすべて実行", m_nID);
70 #if IUTEST_HAS_STD_EMPLACE
71  m_TestCaseList.emplace(m_nID, pTestCase);
72 #else
73  m_TestCaseList.insert( ::std::pair<WORD, const ::iutest::TestCase*>(m_nID, pTestCase) );
74 #endif
75  ++m_nID;
76  for( int j=0; j < test_count; ++j )
77  {
78  const ::iutest::TestInfo* pTestInfo = pTestCase->GetTestInfo(j);
79  Append(hTestCase, pTestInfo->name(), m_nID);
80 #if IUTEST_HAS_STD_EMPLACE
81  m_TestInfoList.emplace(m_nID, pTestInfo);
82 #else
83  m_TestInfoList.insert(::std::pair<WORD, const ::iutest::TestInfo*>(m_nID, pTestInfo));
84 #endif
85  ++m_nID;
86  }
87  }
88  return true;
89  }
90 
91  bool OnCommand(WORD wID)
92  {
93  if( wID == m_nIDTop )
94  {
95  ::iutest::IUTEST_FLAG(filter) = "*";
96  return IUTEST_RUN_ALL_TESTS() == 0 ? true : false;
97  }
98 
99  {
100  TestInfoMap::iterator it = m_TestInfoList.find(wID);
101  if( it != m_TestInfoList.end() )
102  {
103  ::iutest::IUTEST_FLAG(filter) = TestFullName(it->second);
104  return IUTEST_RUN_ALL_TESTS() == 0 ? true : false;
105  }
106  }
107  {
108  TestCaseMap::iterator it = m_TestCaseList.find(wID);
109  if( it != m_TestCaseList.end() )
110  {
111  ::std::string filter = it->second->name();
112  filter += ".*";
113  ::iutest::IUTEST_FLAG(filter) = filter;
114  return IUTEST_RUN_ALL_TESTS() == 0 ? true : false;
115  }
116  }
117  return true;
118  }

121  static bool Append(HMENU hMenu, const char* lpszName, WORD nID)
122  {
123  char str[256];
124  MENUITEMINFOA mii = {0};
125  mii.cbSize = sizeof(mii);
126  mii.fMask = MIIM_ID | MIIM_TYPE;
127  mii.fType = MFT_STRING;
128  mii.dwTypeData = str;
129  mii.wID = nID;
130  strcpy_s(str, lpszName);
131  const int num = ::GetMenuItemCount(hMenu);
132  if( !::InsertMenuItemA(hMenu, num, TRUE, &mii) )
133  {
134  return false;
135  }
136  return true;
137  }
138  static HMENU AppendPopup(HMENU hMenu, const char* lpszName)
139  {
140  HMENU hSubMenu = CreateMenu();
141  char str[256];
142  MENUITEMINFOA mii = {0};
143  mii.cbSize = sizeof(mii);
144  mii.fMask = MIIM_ID | MIIM_TYPE | MIIM_SUBMENU;
145  mii.fType = MFT_STRING;
146  mii.dwTypeData = str;
147  mii.hSubMenu = hSubMenu;
148  strcpy_s(str, lpszName);
149  const int num = ::GetMenuItemCount(hMenu);
150  if( !::InsertMenuItemA(hMenu, num, TRUE, &mii) )
151  {
152  return NULL;
153  }
154  return hSubMenu;
155  }
156 };
